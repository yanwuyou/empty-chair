# 空椅子 MVP 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建「和情绪对话」H5 单文件应用 — 选情绪小怪兽 → 填空表达 → 换椅子回应 → 星星评分。

**Architecture:** 单 HTML 文件，CSS/JS 内嵌。3 个页面通过显示/隐藏 div 切换。LocalStorage 存历史记录。零外部依赖。

**Tech Stack:** HTML5 + CSS3 + 原生 JavaScript（ES6），无框架，无构建工具。

在 `<script>` 中 `selectEmotion` 函数之前添加：

```javascript
  let currentEmotion = null;
  let dialogueStep = 0; // 0=开场, 1=表达, 2=换椅子, 3=回应

  function selectEmotion(emotionId) {
    const all = [...EMOTIONS.negative, ...EMOTIONS.positive];
    currentEmotion = all.find(e => e.id === emotionId);
    dialogueStep = 0;
    renderDialogueStep();
    showPage('dialogue');
  }

  function renderDialogueStep() {
    const monster = document.getElementById('dialogueMonster');
    const bubble = document.getElementById('dialogueBubble');
    const body = document.getElementById('dialogueBody');
    const p = PHRASES[currentEmotion.id];
    
    monster.innerHTML = `
      <span class="emoji-big">${currentEmotion.icon}</span>
      <div class="monster-name" style="color:${currentEmotion.color}">${currentEmotion.name}小怪兽</div>
    `;
    monster.style.transform = 'translateX(0)';
    
    if (dialogueStep === 0) {
      bubble.textContent = p.opening;
      body.innerHTML = `
        <div class="dialogue-step-label">你想对小怪兽说什么？</div>
        <div id="expressionForm"></div>
        <div class="mode-toggle">
          <button class="mode-btn active" onclick="setMode('template')">📋 用模板填</button>
          <button class="mode-btn" onclick="setMode('free')">✍️ 自己写</button>
        </div>
        <button class="btn submit-btn" onclick="submitExpression()">说好了</button>
      `;
      currentMode = 'template';
      renderTemplateForm();
    }
    // Step 2 和 3 在后续 Task 中实现
  }

  let currentMode = 'template';

  function setMode(mode) {
    currentMode = mode;
    document.querySelectorAll('.mode-btn').forEach(b => b.classList.remove('active'));
    event.target.classList.add('active');
    const form = document.getElementById('expressionForm');
    if (mode === 'template') {
      renderTemplateForm();
    } else {
      form.innerHTML = `<textarea class="free-textarea" id="freeText" placeholder="想说什么都可以……不着急，慢慢说。"></textarea>`;
    }
  }

  function renderTemplateForm() {
    const p = PHRASES[currentEmotion.id];
    document.getElementById('expressionForm').innerHTML = `
      <div class="template-form">
        <div class="template-line"><span class="prefix">因为 </span><span class="fill" contenteditable="true" id="fillBecause"></span></div>
        <div class="template-line"><span class="prefix">我感觉 </span><span class="fill" contenteditable="true" id="fillFeel"></span></div>
        <div class="template-line"><span class="prefix">我希望 </span><span class="fill" contenteditable="true" id="fillHope"></span></div>
      </div>
      <div style="text-align:center;margin-top:8px;font-size:13px;color:var(--orange-light);">${p.listening}</div>
    `;
  }
```

```javascript
  let expressionData = null;

  function submitExpression() {
    const p = PHRASES[currentEmotion.id];
    if (currentMode === 'template') {
      const because = document.getElementById('fillBecause')?.textContent?.trim() || '';
      const feel = document.getElementById('fillFeel')?.textContent?.trim() || '';
      const hope = document.getElementById('fillHope')?.textContent?.trim() || '';
      if (!because && !feel && !hope) return;
      expressionData = { because, feel, hope, freeText: null };
    } else {
      const text = document.getElementById('freeText')?.value?.trim() || '';
      if (!text) return;
      expressionData = { because: text, feel: '', hope: '', freeText: text };
    }
    
    dialogueStep = 2; // 换椅子
    const monster = document.getElementById('dialogueMonster');
    const bubble = document.getElementById('dialogueBubble');
    const body = document.getElementById('dialogueBody');

    // 动画：小怪兽右移
    monster.style.transform = 'translateX(60px)';
    monster.style.transition = 'transform 0.4s ease';
    
    bubble.textContent = p.switchChair;
    
    body.innerHTML = `
      <div id="responseForm"></div>
      <div class="mode-toggle">
        <button class="mode-btn active" onclick="setResponseMode('template')">📋 用模板</button>
        <button class="mode-btn" onclick="setResponseMode('free')">✍️ 自己写</button>
      </div>
      <button class="btn submit-btn" onclick="submitResponse()">说好了</button>
    `;
    responseMode = 'template';
    renderResponseTemplate();
  }

  let responseMode = 'template';

  function setResponseMode(mode) {
    responseMode = mode;
    document.querySelectorAll('#dialogueBody .mode-btn').forEach(b => b.classList.remove('active'));
    event.target.classList.add('active');
    if (mode === 'template') renderResponseTemplate();
    else document.getElementById('responseForm').innerHTML = `<textarea class="free-textarea" id="responseFreeText" placeholder="你想对自己说什么？"></textarea>`;
  }

  function renderResponseTemplate() {
    document.getElementById('responseForm').innerHTML = `
      <div class="template-line"><span class="prefix">我想对自己说 </span><span class="fill" contenteditable="true" id="fillResponse"></span></div>
    `;
  }
```

```javascript
  const STORAGE_KEY = 'empty_chair_history';

  function getHistory() {
    try {
      return JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
    } catch(e) {
      return [];
    }
  }

  function saveToHistory() {
    const history = getHistory();
    history.unshift({
      id: Date.now().toString(36) + Math.random().toString(36).slice(2, 6),
      emotionId: currentEmotion.id,
      emotionName: currentEmotion.name,
      emotionIcon: currentEmotion.icon,
      emotionColor: currentEmotion.color,
      date: new Date().toISOString(),
      expression: expressionData,
      response: responseData,
      stars: 0
    });
    localStorage.setItem(STORAGE_KEY, JSON.stringify(history));
  }

  function updateLastHistoryStars(n) {
    const history = getHistory();
    if (history.length > 0) {
      history[0].stars = n;
      localStorage.setItem(STORAGE_KEY, JSON.stringify(history));
    }
  }
```
